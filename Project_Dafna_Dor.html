log.many= function(dat, column.nos) {
    nms= names(dat)
    for(col in column.nos) {
        name= paste(nms[col],".log", sep = "")
        dat[name]= log(dat[,col])
    }
    cat(paste("rescaled with log", length(column.nos), " variable(s)\n"))
    dat
}
